# 11659 : 구간 합 구하기4



## Link

## Code

```java
import java.io.BufferedReader;
import java.io.BufferedWriter;
import java.io.InputStreamReader;
import java.io.OutputStreamWriter;
import java.util.StringTokenizer;
import java.util.stream.Stream;

public class Main {
    public void solution() throws Exception {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        BufferedWriter bw = new BufferedWriter(new OutputStreamWriter(System.out));
        StringTokenizer st = new StringTokenizer(br.readLine());

        int n = Integer.parseInt(st.nextToken());
        int m = Integer.parseInt(st.nextToken());

        int[] array = Stream.of(br.readLine().split(" ")).mapToInt(Integer::parseInt).toArray();

        int[] sumArray = new int[n+1];
        sumArray[0] = 0;
        for (int i=0; i<n; i++) {
            sumArray[i+1] = sumArray[i] + array[i];
        }

        for (int j=0; j<m; j++) {
            int[] prefix = Stream.of(br.readLine().split(" ")).mapToInt(Integer::parseInt).toArray();
            bw.write(String.valueOf(sumArray[prefix[1]] - sumArray[prefix[0]-1]));
            bw.newLine();
        }

        bw.flush();
        br.close();
        bw.close();

    }

    public static void main(String[] args) throws Exception {
        new Main().solution();
    }
}
```
